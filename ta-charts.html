OpenClaw Chart Config

Chart settings live in ta-config.yaml. Set charts.format to html for interactive Plotly charts or png for email-friendly batch exports. The charts.tickers list defaults to your full watchlist if not specified.

Python 
import yaml

def load_config(path: str = "ta-config.yaml") -> dict:
    with open(path) as f:
        return yaml.safe_load(f)

cfg      = load_config()
chart_cfg = cfg.get("charts", {})

ENABLED    = chart_cfg.get("enabled", True)
FORMAT     = chart_cfg.get("format", "html")       # "html" or "png"
OUTPUT_DIR = chart_cfg.get("output_dir", "charts/")
CHART_TICKERS = chart_cfg.get("tickers") or cfg["watchlist"]["tickers"]

print(f"🦞 OpenClaw Charts | {len(CHART_TICKERS)} tickers | Format: {FORMAT} | Dir: {OUTPUT_DIR}")

Two Approaches

You have two main paths to automated charting. Choose based on your workflow:

Plotly: Interactive HTML Files

Best for: Daily browser review, exploration, and client presentations.

Each chart is a standalone HTML file (~200KB) you can open in any browser. Zoom, pan, and hover to inspect exact values. Great for your main watchlist (5–15 tickers). Single files or multi-chart dashboards available. Charts remain interactive indefinitely.

matplotlib: Static PNG/SVG/PDF

Best for: Batch reports, email delivery, and archive snapshots.

Lightweight PNG images (~150KB) perfect for sending in emails or embedding in daily reports. Faster to generate for large batches (20+ tickers). No interactive features, but crisp, print-ready output. Ideal for nightly overnight runs and scheduled reports.

Rule of thumb: Use Plotly for your main watchlist; use matplotlib for batch reports over 20+ tickers.

Plotly Candlestick with MA Overlays

Build a 3-panel interactive chart: candlesticks + moving averages + Bollinger Bands, volume bars, and RSI with overbought/oversold zones.

Python 
# pip install plotly yfinance pandas-ta
import plotly.graph_objects as go
from plotly.subplots import make_subplots
import yfinance as yf
import pandas_ta as ta
import pandas as pd

def fetch_ohlcv(ticker: str, period: str = "6mo") -> pd.DataFrame:
    df = yf.download(ticker, period=period, interval="1d", progress=False)
    df.columns = [c.lower() for c in df.columns]
    return df[["open","high","low","close","volume"]].dropna()

def plotly_chart(ticker: str, period: str = "6mo", save_path: str = None) -> go.Figure:
    """
    Build a 3-panel Plotly chart:
    Panel 1: Candlestick + SMA20/50/200 + Bollinger Bands
    Panel 2: Volume bars
    Panel 3: RSI with 30/70 lines
    """
    df = fetch_ohlcv(ticker, period)

    # Indicators
    df["sma20"]  = ta.sma(df["close"], 20)
    df["sma50"]  = ta.sma(df["close"], 50)
    df["sma200"] = ta.sma(df["close"], 200)
    df["rsi"]    = ta.rsi(df["close"], 14)
    bb = ta.bbands(df["close"], 20, 2)
    if bb is not None:
        bbu_col = [c for c in bb.columns if "BBU" in c][0]
        bbl_col = [c for c in bb.columns if "BBL" in c][0]
        df["bb_upper"] = bb[bbu_col]
        df["bb_lower"] = bb[bbl_col]

    fig = make_subplots(
        rows=3, cols=1,
        shared_xaxes=True,
        vertical_spacing=0.03,
        row_heights=[0.60, 0.15, 0.25],
        subplot_titles=(f"{ticker} — Price", "Volume", "RSI(14)")
    )

    # Candlesticks
    fig.add_trace(go.Candlestick(
        x=df.index, open=df["open"], high=df["high"],
        low=df["low"], close=df["close"],
        name="Price",
        increasing_line_color="#22c55e",
        decreasing_line_color="#ef4444"
    ), row=1, col=1)

    # Moving averages
    for col, color, name in [
        ("sma20", "#fbbf24", "SMA 20"),
        ("sma50", "#60a5fa", "SMA 50"),
        ("sma200","#f472b6", "SMA 200"),
    ]:
        fig.add_trace(go.Scatter(
            x=df.index, y=df[col],
            name=name, line=dict(color=color, width=1.2),
            opacity=0.8
        ), row=1, col=1)

    # Bollinger Bands
    if "bb_upper" in df.columns:
        fig.add_trace(go.Scatter(
            x=df.index, y=df["bb_upper"],
            name="BB Upper", line=dict(color="#0d9488", width=1, dash="dot"),
            opacity=0.6
        ), row=1, col=1)
        fig.add_trace(go.Scatter(
            x=df.index, y=df["bb_lower"],
            name="BB Lower", line=dict(color="#0d9488", width=1, dash="dot"),
            fill="tonexty", fillcolor="rgba(13,148,136,0.06)",
            opacity=0.6
        ), row=1, col=1)

    # Volume
    colors = ["#22c55e" if c >= o else "#ef4444"
              for c, o in zip(df["close"], df["open"])]
    fig.add_trace(go.Bar(
        x=df.index, y=df["volume"],
        name="Volume", marker_color=colors, opacity=0.7
    ), row=2, col=1)

    # RSI
    fig.add_trace(go.Scatter(
        x=df.index, y=df["rsi"],
        name="RSI", line=dict(color="#a78bfa", width=1.5)
    ), row=3, col=1)
    fig.add_hline(y=70, line_dash="dot", line_color="#ef4444", opacity=0.5, row=3, col=1)
    fig.add_hline(y=30, line_dash="dot", line_color="#22c55e", opacity=0.5, row=3, col=1)

    fig.update_layout(
        template="plotly_dark",
        title=f"{ticker} | {period} | Generated {pd.Timestamp.now().strftime('%Y-%m-%d')}",
        height=800,
        showlegend=True,
        xaxis_rangeslider_visible=False,
        font=dict(family="system-ui, sans-serif"),
        paper_bgcolor="#1c1917",
        plot_bgcolor="#1c1917",
    )
    fig.update_xaxes(showgrid=True, gridcolor="#292524")
    fig.update_yaxes(showgrid=True, gridcolor="#292524")

    if save_path:
        fig.write_html(save_path)
        print(f"Saved: {save_path}")

    return fig

# Generate and save
fig = plotly_chart("AAPL", period="6mo", save_path="charts/AAPL.html")

Batch Plotly Chart Packet

Generate an HTML chart for each ticker and a master index page linking them all. Perfect for organizing a full watchlist.

Python 
import os

def generate_chart_packet(tickers: list, output_dir: str = "charts", period: str = "6mo") -> None:
    """
    Generate an HTML chart for each ticker and a master index page.
    """
    os.makedirs(output_dir, exist_ok=True)
    generated = []

    for ticker in tickers:
        try:
            path = os.path.join(output_dir, f"{ticker}.html")
            plotly_chart(ticker, period=period, save_path=path)
            generated.append(ticker)
            print(f"  ✓ {ticker}")
        except Exception as e:
            print(f"  ✗ {ticker}: {e}")

    # Write index.html linking to all charts
    links = "\n".join(
        f'
  • {t}
    • '
        for t in generated
    )
    index_html = f"""

Chart Packet — {pd.Timestamp.now().strftime('%Y-%m-%d')}




    🦞 OpenClaw Chart Packet
    

    Generated: {pd.Timestamp.now().strftime('%Y-%m-%d %H:%M')}
    

      {links}
    
"""

    with open(os.path.join(output_dir, "index.html"), "w") as f:
        f.write(index_html)

    print(f"\nPacket complete: {len(generated)}/{len(tickers)} charts")
    print(f"Open: {output_dir}/index.html")

WATCHLIST = ["AAPL","MSFT","GOOGL","AMZN","NVDA","META","SPY","QQQ","GLD","TLT"]
generate_chart_packet(WATCHLIST, output_dir="charts", period="6mo")

    matplotlib PNG Chart for Email-Friendly Batches

    Generate lightweight, email-ready PNG charts with candlesticks, moving averages, volume, and RSI. Perfect for automated daily reports.

    Python 
    import matplotlib.pyplot as plt
import matplotlib.dates as mdates
from matplotlib.patches import Rectangle

def matplotlib_chart(ticker: str, period: str = "6mo", save_path: str = None) -> None:
    """
    3-panel matplotlib chart: price+MAs, volume, RSI.
    Saves as PNG — email-friendly, lightweight.
    """
    df = fetch_ohlcv(ticker, period)
    df["sma20"]  = ta.sma(df["close"], 20)
    df["sma50"]  = ta.sma(df["close"], 50)
    df["sma200"] = ta.sma(df["close"], 200)
    df["rsi"]    = ta.rsi(df["close"], 14)

    fig, (ax1, ax2, ax3) = plt.subplots(
        3, 1, figsize=(14, 9),
        gridspec_kw={"height_ratios": [3, 1, 1.2]},
        facecolor="#1c1917"
    )
    for ax in (ax1, ax2, ax3):
        ax.set_facecolor("#1c1917")
        ax.tick_params(colors="#a8a29e")
        ax.spines[:].set_color("#44403c")

    # Candlesticks (simplified as line + color fill)
    up   = df[df["close"] >= df["open"]]
    down = df[df["close"] < df["open"]]
    ax1.vlines(up.index,   up["low"],   up["high"],   color="#22c55e", lw=0.8)
    ax1.vlines(down.index, down["low"], down["high"], color="#ef4444", lw=0.8)
    ax1.vlines(up.index,   up["open"],  up["close"],  color="#22c55e", lw=3)
    ax1.vlines(down.index, down["open"],down["close"],color="#ef4444", lw=3)

    ax1.plot(df.index, df["sma20"],  color="#fbbf24", lw=1,   label="SMA20",  alpha=0.9)
    ax1.plot(df.index, df["sma50"],  color="#60a5fa", lw=1.2, label="SMA50",  alpha=0.9)
    ax1.plot(df.index, df["sma200"], color="#f472b6", lw=1.2, label="SMA200", alpha=0.9)
    ax1.set_title(f"{ticker}  |  {period}  |  {pd.Timestamp.now().strftime('%Y-%m-%d')}",
                  color="#e7e5e4", fontsize=13)
    ax1.legend(facecolor="#292524", edgecolor="#44403c", labelcolor="#e7e5e4", fontsize=8)
    ax1.xaxis.set_major_formatter(mdates.DateFormatter("%b '%y"))

    # Volume
    vol_colors = ["#22c55e" if c >= o else "#ef4444"
                  for c, o in zip(df["close"], df["open"])]
    ax2.bar(df.index, df["volume"], color=vol_colors, alpha=0.7, width=0.8)
    ax2.set_ylabel("Vol", color="#a8a29e", fontsize=8)

    # RSI
    ax3.plot(df.index, df["rsi"], color="#a78bfa", lw=1.5)
    ax3.axhline(70, color="#ef4444", ls="--", lw=0.8, alpha=0.7)
    ax3.axhline(30, color="#22c55e", ls="--", lw=0.8, alpha=0.7)
    ax3.fill_between(df.index, df["rsi"], 70, where=df["rsi"]>70, alpha=0.15, color="#ef4444")
    ax3.fill_between(df.index, df["rsi"], 30, where=df["rsi"]<30, alpha=0.15, color="#22c55e")
    ax3.set_ylim(0, 100)
    ax3.set_ylabel("RSI", color="#a8a29e", fontsize=8)

    plt.tight_layout(pad=1.5)

    path = save_path or f"charts/{ticker}.png"
    os.makedirs(os.path.dirname(path), exist_ok=True)
    plt.savefig(path, dpi=150, bbox_inches="tight", facecolor="#1c1917")
    plt.close()
    print(f"Saved PNG: {path}")

matplotlib_chart("MSFT", period="6mo", save_path="charts/MSFT.png")

    Nightly Cron Job: Full Pipeline Automation

    Automate the complete workflow: fetch fresh data, generate charts, and deliver your chart packet. Schedule with cron to run Monday–Friday at 5 PM.

    Config-driven nightly run — reads all settings from ta-config.yaml:

    Python 
    # Config-driven nightly run — reads from ta-config.yaml
def openclaw_nightly_charts() -> None:
    cfg      = load_config()
    chart_cfg = cfg.get("charts", {})

    if not chart_cfg.get("enabled", True):
        print("Charts disabled in ta-config.yaml — skipping.")
        return

    fmt      = chart_cfg.get("format", "html")
    out_dir  = chart_cfg.get("output_dir", "charts/")
    tickers  = chart_cfg.get("tickers") or cfg["watchlist"]["tickers"]
    period   = cfg["scanner"]["lookback_period"]

    print(f"\n🦞 OpenClaw Nightly Chart Run | {len(tickers)} tickers | {fmt.upper()}")

    if fmt == "png":
        for t in tickers:
            matplotlib_chart(t, period=period, save_path=f"{out_dir}{t}.png")
    else:
        generate_chart_packet(tickers, output_dir=out_dir, period=period)

    print(f"\n✓ Charts ready: {out_dir}index.html")

# Schedule via cron — runs Mon–Fri at 5:00pm
# 0 17 * * 1-5 python3 /path/to/openclaw/nightly_charts.py
openclaw_nightly_charts()

    Legacy non-config version (still supported):

    Python 
    from datetime import date
import subprocess

def nightly_chart_run(watchlist: list, output_dir: str = None) -> None:
    """
    Full nightly pipeline:
    1. Fetch fresh data
    2. Generate Plotly charts
    3. Run scanner and annotate results
    4. Write chart packet index
    """
    today = date.today().isoformat()
    out = output_dir or f"chart_packets/{today}"
    os.makedirs(out, exist_ok=True)

    print(f"{'🦞 Nightly Chart Run':=^50}")
    print(f"Date: {today} | Tickers: {len(watchlist)}\n")

    generate_chart_packet(watchlist, output_dir=out, period="6mo")

    print(f"\nDone. Open: file://{os.path.abspath(out)}/index.html")

# Schedule with cron (runs Mon–Fri at 5pm):
# 0 17 * * 1-5 python3 /home/user/openclaw/nightly_charts.py
WATCHLIST = ["AAPL","MSFT","GOOGL","NVDA","META","AMZN","SPY","QQQ","IWM","GLD"]
nightly_chart_run(WATCHLIST)
